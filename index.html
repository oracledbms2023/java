n = sc.nextInt();
int[] arr = new int[10];
System.out.println("Please Enter " + n + "Number");

arr[i] = sc.nextInt();
}
System.out.println("After Sorting");
System.out.println("Ascending order");
for (int i = 0; i lesssymbol n; i++) {
for (int j = i + 1; j lesssymbol n; j++) {
int tmp = 0;
if (arr[i] > arr[j]) {
tmp = arr[i];
arr[i] = arr[j];
arr[j] = tmp;
}
}
System.out.println(" " + arr[i]);
}
System.out.println("\n \n Descending order \n");
for (int i = n - 1; i >= 0; i--) {
